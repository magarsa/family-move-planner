Computed values (derived, not stored)
const downAmt = price * (downPct / 100)
const loan = price - downAmt
const monthlyRate = rate / 100 / 12
const n = term * 12
const pAndI = loan * (monthlyRate * (1 + monthlyRate) ** n) / ((1 + monthlyRate) ** n - 1)
const monthlyTax = price * (taxRate / 100) / 12
const monthlyIns = annualInsurance / 12
const monthlyPMI = downPct < 20 ? (loan * (pmiRate / 100) / 12) : 0
const totalMonthly = pAndI + monthlyTax + monthlyIns + monthlyPMI
const grossMonthlyIncome = annualIncome / 12

// Scenario B: rental income offsets Des Moines mortgage in lender DTI calc
// Lender credits 75% of rent toward offsetting Des Moines PITI (~$1,900/mo)
const desMoinesNetDTIExposure = scenario === 'B'
? Math.max(0, 1900 - desMoinesRent * 0.75)
: 0
const monthlyDebt = totalMonthly + desMoinesNetDTIExposure
const dti = (monthlyDebt / grossMonthlyIncome) * 100

// Buydown savings (2-1 buydown at current rate)
const year1Rate = Math.max(0, rate - 2)
const year2Rate = Math.max(0, rate - 1)
const year1PI = loan * ((year1Rate/100/12) * (1 + year1Rate/100/12)**n) / ((1 + year1Rate/100/12)**n - 1)
const year2PI = loan * ((year2Rate/100/12) * (1 + year2Rate/100/12)**n) / ((1 + year2Rate/100/12)**n - 1)
const year1Savings = pAndI - year1PI
const year2Savings = pAndI - year2PI
